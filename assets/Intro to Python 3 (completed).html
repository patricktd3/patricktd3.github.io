# For a system, we can use essentially the same syntax.
# The trick is that instead of just passing in a single value of y, we now need to pass in a vector X = [x1, x2].
# This also means that our derivative function will have to take and return the same vector X.

# Here we define the derivative function for our system
def deriv(X,t):
    x1 = X[0] - 5*X[1]
    x2 = X[0] - 3*X[1]
    return np.array([x1,x2])

# Then set up a t vector and solve the system for a given initial condition 
tt = np.linspace(0,8,1000)

X0 = [-1,1] # initial condition x1(0) = -1 and x2(0) = 1
XX = odeint(deriv,X0,tt)

# The following function will generate a phase diagram. The syntax is:
# phase(<der func>,<xrange>,<yrange>,<meshdensity>)
#
# where:
# <der func> is the function that calculates the derivative
# <xrange> is the range of x1 values
# <yrange> is the range of x2 values
# <meshdensity> determines the number of arrows

def phase(f,xrange,yrange):    
    # BUILD A 20x20 GRID
    xx,yy = np.meshgrid(np.linspace(xrange[0],xrange[1],20),np.linspace(yrange[0],yrange[1],20))
    
    # FIND THE ARROWS AT THE GRID POINTS
    dxx,dyy = f([xx,yy],0) # assuming an autonomous system
    
    # MAKE THE ARROWS UNIT LENGTH
    dxxU = np.divide(dxx,np.sqrt(np.square(dxx) + np.square(dyy)))
    dyyU = np.divide(dyy,np.sqrt(np.square(dxx) + np.square(dyy)))
    
    # PLOT IT
    plt.figure(figsize=(8,8)) # figsize=(#1,#2) will generate a figure with that size
    plt.plot([xrange[0],xrange[1]],[0,0],'k--',linewidth=2)
    plt.plot([0,0],[yrange[0],yrange[1]],'k--',linewidth=2)
    plt.quiver(xx,yy,dxxU,dyyU,color='b',alpha=0.75) # alpha controls how dark the arrows are
    plt.xlabel('x1')
    plt.ylabel('x2')
    plt.axis([xrange[0],xrange[1],yrange[0],yrange[1]])
    plt.title('Phase Diagram')
    plt.show()

# With this, we can define the derivative function for the system
def deriv3(X,t):
    x1 = X[1]
    x2 = 4*np.cos(2*t) - 5*X[1] - 6*X[0]
    return np.array([x1,x2])

# Then set up a t vector and solve the system for a given initial condition 
tt = np.linspace(0,8,100)

X0 = [1,0] # initial conditions
XX = odeint(deriv3,X0,tt)

# Pull off u = x1
uu = XX[:,0]

# And plot it along with the analytic solution
def phi(t):
    return 2*np.exp(-2*t) - 14/13*np.exp(-3*t) + 1/13*np.cos(2*t) + 5/13*np.sin(2*t)

plt.figure()
plt.plot(tt,uu,label='odeint()',linewidth=2)
plt.plot(tt,phi(tt),'--k',label='analytic',linewidth=2)
plt.title('Spring Displacement')
plt.xlabel('t')
plt.legend()
plt.show()
