# DEFINE A FUNCTION TO IMPLEMENT EULER'S METHOD
def euler(f,y0,a,b,h):
    # Pre-Allocate
    numsteps = int(math.ceil((b-a)/h))
    t, y = [None] * (numsteps+1), [None] * (numsteps+1)
    
    # Initialize
    t[0],y[0] = a,y0
    
    # Run Euler's Method
    for j in range(1,numsteps+1):
        t[j] = t[j-1] + h
        y[j] = y[j-1] + h*f(y[j-1], t[j-1])
    
    return (np.asarray(t),np.asarray(y))
