# The following function will generate a phase diagram. The syntax is:
# phase(<der func>,<xrange>,<yrange>,<meshdensity>)
#
# where:
# <der func> is the function that calculates the derivative
# <xrange> is the range of x1 values
# <yrange> is the range of x2 values
# <meshdensity> determines the number of arrows

def phase(f,xrange,yrange):    
    # BUILD A 20x20 GRID
    xx,yy = np.meshgrid(np.linspace(xrange[0],xrange[1],20),np.linspace(yrange[0],yrange[1],20))
    
    # FIND THE ARROWS AT THE GRID POINTS
    dxx,dyy = f([xx,yy],0) # assuming an autonomous system
    
    # MAKE THE ARROWS UNIT LENGTH
    dxxU = np.divide(dxx,np.sqrt(np.square(dxx) + np.square(dyy)))
    dyyU = np.divide(dyy,np.sqrt(np.square(dxx) + np.square(dyy)))
    
    # PLOT IT
    plt.figure(figsize=(8,8))
    plt.plot([xrange[0],xrange[1]],[0,0],'k--',linewidth=2)
    plt.plot([0,0],[yrange[0],yrange[1]],'k--',linewidth=2)
    plt.quiver(xx,yy,dxxU,dyyU,color='b',alpha=0.75) # alpha controls how dark the arrows are
    plt.xlabel('x1')
    plt.ylabel('x2')
    plt.axis([xrange[0],xrange[1],yrange[0],yrange[1]])
    plt.title('Phase Diagram')
    plt.show()
